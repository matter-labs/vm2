14/// Heap page size in bytes.
15const HEAP_PAGE_SIZE: usize = 1 << 12;

17/// Heap page.
18#[derive(Debug, Clone, PartialEq)]
19struct HeapPage(Box<[u8; HEAP_PAGE_SIZE]>);

21impl Default for HeapPage {
22    fn default() -> Self {
23        let boxed_slice: Box<[u8]> = vec![0_u8; HEAP_PAGE_SIZE].into();
24        Self(boxed_slice.try_into().unwrap())
25    }

33// The reference VM treats reads from missing memory pages as reads from an
34// all-zero page. Keep write paths strict, but make read-only indexing total so
35// panic-produced or otherwise empty fat pointers cannot abort the host.
36static EMPTY_HEAP: Heap = Heap { pages: Vec::new() };
37
38// We never remove `HeapPage`s (even after rollbacks – although we do zero all added pages in this case),
39// we allow additional pages to be present if they are zeroed.
40impl PartialEq for Heap {
41    fn eq(&self, other: &Self) -> bool {
42        for i in 0..self.pages.len().max(other.pages.len()) {
43            let this_page = self.pages.get(i).and_then(Option::as_ref);
44            let other_page = other.pages.get(i).and_then(Option::as_ref);
45            match (this_page, other_page) {
46                (Some(this_page), Some(other_page)) => {
47                    if this_page != other_page {
48                        return false;
49                    }
50                }
51                (Some(page), None) | (None, Some(page)) => {
52                    if page.0.iter().any(|&byte| byte != 0) {
53                        return false;
54                    }
55                }
56                (None, None) => { /* do nothing */ }
57            }
58        }
59        true
60    }
61}

63impl Heap {
64    fn from_bytes(bytes: &[u8], pagepool: &mut PagePool) -> Self {
65        let pages = bytes
66            .chunks(HEAP_PAGE_SIZE)
67            .map(|bytes| {
68                Some(if let Some(mut page) = pagepool.get_dirty_page() {
69                    page.0[..bytes.len()].copy_from_slice(bytes);
70                    page.0[bytes.len()..].fill(0);
71                    page
72                } else {
73                    let mut page = HeapPage::default();
74                    page.0[..bytes.len()].copy_from_slice(bytes);
75                    page
76                })
77            })
78            .collect();
79        Self { pages }
80    }
81
82    fn recycle(self, pagepool: &mut PagePool) {
83        for page in self.pages.into_iter().flatten() {
84            pagepool.recycle_page(page);
85        }
86    }

88    pub(crate) fn read_u256(&self, start_address: u32) -> U256 {
89        let (page_idx, offset_in_page) = address_to_page_offset(start_address);
90        let bytes_in_page = HEAP_PAGE_SIZE - offset_in_page;
91
92        if bytes_in_page >= 32 {
93            if let Some(page) = self.page(page_idx) {
94                U256::from_big_endian(&page.0[offset_in_page..offset_in_page + 32])
95            } else {
96                U256::zero()
97            }
98        } else {
99            let mut result = [0u8; 32];
100            if let Some(page) = self.page(page_idx) {
101                for (res, src) in result.iter_mut().zip(&page.0[offset_in_page..]) {
102                    *res = *src;
103                }
104            }
105            if let Some(page) = self.page(page_idx + 1) {
106                for (res, src) in result[bytes_in_page..].iter_mut().zip(&*page.0) {
107                    *res = *src;
108                }
109            }
110            U256::from_big_endian(&result)
111        }
112    }

114    pub(crate) fn read_u256_partially(&self, range: Range<u32>) -> U256 {
115        let (page_idx, offset_in_page) = address_to_page_offset(range.start);
116        let length = range.len();
117        let bytes_in_page = length.min(HEAP_PAGE_SIZE - offset_in_page);
118
119        let mut result = [0u8; 32];
120        if let Some(page) = self.page(page_idx) {
121            for (res, src) in result[..bytes_in_page]
122                .iter_mut()
123                .zip(&page.0[offset_in_page..])
124            {
125                *res = *src;
126            }
127        }
128        if let Some(page) = self.page(page_idx + 1) {
129            for (res, src) in result[bytes_in_page..length].iter_mut().zip(&*page.0) {
130                *res = *src;
131            }
132        }
133        U256::from_big_endian(&result)
134    }

136    pub(crate) fn read_range_big_endian(&self, range: Range<u32>) -> Vec<u8> {
137        let length = range.len();
138
139        let (mut page_idx, mut offset_in_page) = address_to_page_offset(range.start);
140        let mut result = Vec::with_capacity(length);
141        while result.len() < length {
142            let len_in_page = (length - result.len()).min(HEAP_PAGE_SIZE - offset_in_page);
143            if let Some(page) = self.page(page_idx) {
144                result.extend_from_slice(&page.0[offset_in_page..(offset_in_page + len_in_page)]);
145            } else {
146                result.resize(result.len() + len_in_page, 0);
147            }
148            page_idx += 1;
149            offset_in_page = 0;
150        }
151        result
152    }

155    pub(crate) fn read_byte(&self, address: u32) -> u8 {
156        let (page, offset) = address_to_page_offset(address);
157        self.page(page).map_or(0, |page| page.0[offset])

164    fn get_or_insert_page(&mut self, idx: usize, pagepool: &mut PagePool) -> &mut HeapPage {
165        if self.pages.len() <= idx {
166            self.pages.resize(idx + 1, None);
167        }
168        self.pages[idx].get_or_insert_with(|| pagepool.allocate_page())
169    }

171    fn write_u256(&mut self, start_address: u32, value: U256, pagepool: &mut PagePool) {
172        let (page_idx, offset_in_page) = address_to_page_offset(start_address);
173        let bytes_in_page = HEAP_PAGE_SIZE - offset_in_page;
174        let page = self.get_or_insert_page(page_idx, pagepool);
175
176        if bytes_in_page >= 32 {
177            value.to_big_endian(&mut page.0[offset_in_page..offset_in_page + 32]);
178        } else {
179            let mut bytes = [0; 32];
180            value.to_big_endian(&mut bytes);
181            let mut bytes_iter = bytes.into_iter();
182
183            for (dst, src) in page.0[offset_in_page..].iter_mut().zip(bytes_iter.by_ref()) {
184                *dst = src;
185            }
186
187            let page = self.get_or_insert_page(page_idx + 1, pagepool);
188            for (dst, src) in page.0.iter_mut().zip(bytes_iter) {
189                *dst = src;
190            }
191        }
192    }
193
194    fn write_bytes(&mut self, start_address: u32, bytes: &[u8], pagepool: &mut PagePool) {
195        let (mut page_idx, mut offset_in_page) = address_to_page_offset(start_address);
196        let mut remaining = bytes;
197        while !remaining.is_empty() {
198            let bytes_in_page = (HEAP_PAGE_SIZE - offset_in_page).min(remaining.len());
199            let page = self.get_or_insert_page(page_idx, pagepool);
200            page.0[offset_in_page..offset_in_page + bytes_in_page]
201                .copy_from_slice(&remaining[..bytes_in_page]);
202            remaining = &remaining[bytes_in_page..];
203            page_idx += 1;
204            offset_in_page = 0;
205        }
206    }
207}

209#[inline(always)]
210fn address_to_page_offset(address: u32) -> (usize, usize) {
211    let offset = address as usize;
212    (offset >> 12, offset & (HEAP_PAGE_SIZE - 1))
213}

269impl DecodedPage {
270    fn decode(page: HeapId) -> Option<Self> {
271        if page == static_memory_page() {
272            return Some(Self::Static);
273        }
274        if page == bootloader_calldata_page() {
275            return Some(Self::BootloaderCalldata);
276        }
277        if page == bootloader_heap_page() {
278            return Some(Self::BootloaderHeap);
279        }
280        if page == bootloader_aux_heap_page() {
281            return Some(Self::BootloaderAuxHeap);
282        }
283
284        let raw = page.as_u32();
285        if raw < STARTING_BASE_PAGE {
286            return None;
287        }
288
289        let rel = raw - STARTING_BASE_PAGE;
290        let group = usize::try_from(rel / NEW_MEMORY_PAGES_PER_FAR_CALL).unwrap();
291        match rel % NEW_MEMORY_PAGES_PER_FAR_CALL {
292            0 => Some(Self::Dynamic {
293                group,
294                kind: DynamicPageKind::Code,
295            }),
296            2 => Some(Self::Dynamic {
297                group,
298                kind: DynamicPageKind::Heap,
299            }),
300            3 => Some(Self::Dynamic {
301                group,
302                kind: DynamicPageKind::Aux,
303            }),
304            _ => None,
305        }
306    }
307
308    const fn is_always_allocated(self) -> bool {
309        matches!(
310            self,
311            Self::Static
312                | Self::BootloaderCalldata
313                | Self::BootloaderHeap
314                | Self::BootloaderAuxHeap
315        )
316    }
317}

351    pub(crate) fn allocate_with_content_at(&mut self, page: HeapId, memory: &[u8]) -> HeapId {
352        let decoded = DecodedPage::decode(page)
353            .unwrap_or_else(|| panic!("heap page {} is not decodable", page.as_u32()));
354
355        assert!(
356            !decoded.is_always_allocated(),
357            "heap page {} is already allocated",
358            page.as_u32()
359        );
360
361        let slot = decoded_dynamic_slot_mut(&mut self.dynamic, decoded);
362        // `decoded_page_mut_for_write` populates dynamic slots lazily, so in principle a prior
363        // write could materialize this slot. In production, `allocate_at` is only called from
364        // far-call setup on the heap/aux slots of a freshly assigned base group, which no
365        // prior code path writes to. If this fires, that invariant has been broken.
366        assert!(
367            slot.is_none(),
368            "heap page {} is already allocated",
369            page.as_u32()
370        );
371        *slot = Some(Heap::from_bytes(memory, &mut self.pagepool));
372        page
373    }

375    #[cfg(test)]
376    pub(crate) fn contains(&self, page: HeapId) -> bool {
377        DecodedPage::decode(page).is_some_and(|decoded| {
378            decoded.is_always_allocated() || self.try_decoded_page(decoded).is_some()
379        })
380    }
381
382    // The three panics in this function flag VM-internal bookkeeping bugs, not
383    // reachable conditions: the VM only deallocates pages it previously allocated via
384    // `allocate_at` (frame pop in `pop_frame`, snapshot rollback in `State::rollback`). The
385    // reference `zk_evm` has no deallocation path of its own, so there is no behaviour to
386    // diverge from here — these panics remain as fail-fast diagnostics for our own bookkeeping.
387    pub(crate) fn deallocate(&mut self, page: HeapId) {
388        let decoded = DecodedPage::decode(page)
389            .unwrap_or_else(|| panic!("heap page {} is not decodable", page.as_u32()));
390
391        assert!(
392            !decoded.is_always_allocated(),
393            "heap page {} must remain allocated",
394            page.as_u32()
395        );
396
397        let heap = decoded_dynamic_slot_mut(&mut self.dynamic, decoded)
398            .take()
399            .unwrap_or_else(|| panic!("heap page {} is not allocated", page.as_u32()));
400        heap.recycle(&mut self.pagepool);
401    }

407    pub(crate) fn write_u256(&mut self, page: HeapId, start_address: u32, value: U256) {
408        self.record_bootloader_word_rollback(page, start_address);
409        // Current callers pass `HeapId`s from VM-controlled sources: store opcodes (current
410        // frame heap/aux), static memory, and kernel-gated precompile output. The
411        // `StateInterface::write_heap_u256` tracer entry can in principle pass an arbitrary
412        // `HeapId`. The reference `zk_evm` would silently materialize any page number; we
413        // keep the panic as a tripwire against any new code path.
414        let decoded = DecodedPage::decode(page)
415            .unwrap_or_else(|| panic!("heap page {} is not decodable", page.as_u32()));
416        let (heap, pagepool) = self.decoded_page_mut_for_write(decoded);
417        heap.write_u256(start_address, value, pagepool);
418    }
419
420    pub(crate) fn write_bytes(&mut self, page: HeapId, start_address: u32, bytes: &[u8]) {
421        self.record_bootloader_range_rollback(page, start_address, bytes.len());
422        // Same tripwire rationale as `write_u256`: the only production callers go through
423        // `materialize_decommit_page` (a fresh code page on far call, or the current frame
424        // heap from the `Decommit` opcode), both supplying decodable `HeapId`s. An
425        // undecodable page here indicates a new code path that should be reviewed before
426        // merging.
427        let decoded = DecodedPage::decode(page)
428            .unwrap_or_else(|| panic!("heap page {} is not decodable", page.as_u32()));
429        let (heap, pagepool) = self.decoded_page_mut_for_write(decoded);
430        heap.write_bytes(start_address, bytes, pagepool);
431    }

440    pub(crate) fn rollback(&mut self, (heap_snap, aux_snap): (usize, usize)) {
441        for (address, value) in self.bootloader_heap_rollback_info.drain(heap_snap..).rev() {
442            self.bootloader_heap
443                .write_u256(address, value, &mut self.pagepool);
444        }

447            self.bootloader_aux_heap
448                .write_u256(address, value, &mut self.pagepool);
449        }

470    fn record_bootloader_word_rollback(&mut self, page: HeapId, start_address: u32) {
471        if page == HeapId::FIRST {
472            let prev_value = self[page].read_u256(start_address);
473            self.bootloader_heap_rollback_info
474                .push((start_address, prev_value));
475        } else if page == HeapId::FIRST_AUX {
476            let prev_value = self[page].read_u256(start_address);
477            self.bootloader_aux_rollback_info
478                .push((start_address, prev_value));
479        }
480    }
481
482    fn record_bootloader_range_rollback(&mut self, page: HeapId, start_address: u32, len: usize) {
483        if len == 0 {
484            return;
485        }
486
487        let start = usize::try_from(start_address).expect("heap write address overflow");
488        let end = start.checked_add(len).expect("heap write range overflow");
489        let first_word = start / 32 * 32;
490        let last_word = (end - 1) / 32 * 32;
491
492        for address in (first_word..=last_word).step_by(32) {
493            self.record_bootloader_word_rollback(
494                page,
495                u32::try_from(address).expect("heap write address overflow"),
496            );
497        }
498    }

500    fn try_decoded_page(&self, page: DecodedPage) -> Option<&Heap> {
501        match page {
502            DecodedPage::Static => Some(&self.static_memory),
503            DecodedPage::BootloaderCalldata => Some(&self.bootloader_calldata),
504            DecodedPage::BootloaderHeap => Some(&self.bootloader_heap),
505            DecodedPage::BootloaderAuxHeap => Some(&self.bootloader_aux_heap),
506            DecodedPage::Dynamic { group, kind } => {
507                self.dynamic.get(group).and_then(|group| group.slot(kind))
508            }
509        }
510    }
511
512    fn decoded_page(&self, page: DecodedPage) -> &Heap {
513        self.try_decoded_page(page).unwrap_or(&EMPTY_HEAP)
514    }
515
516    fn decoded_page_mut_for_write(&mut self, page: DecodedPage) -> (&mut Heap, &mut PagePool) {
517        let Self {
518            static_memory,
519            bootloader_calldata,
520            bootloader_heap,
521            bootloader_aux_heap,
522            dynamic,
523            pagepool,
524            ..
525        } = self;
526
527        // The reference memory model materializes a page before writing to it.
528        // vm2 keeps the stricter page-id classification, but decodable dynamic
529        // pages are valid write targets and should be allocated lazily.
530        let heap = match page {
531            DecodedPage::Static => static_memory,
532            DecodedPage::BootloaderCalldata => bootloader_calldata,
533            DecodedPage::BootloaderHeap => bootloader_heap,
534            DecodedPage::BootloaderAuxHeap => bootloader_aux_heap,
535            DecodedPage::Dynamic { group, kind } => {
536                dynamic_slot_mut(dynamic, group, kind).get_or_insert_with(Heap::default)
537            }
538        };
539        (heap, pagepool)
540    }

554impl PartialEq for Heaps {
555    fn eq(&self, other: &Self) -> bool {
556        if self.static_memory != other.static_memory
557            || self.bootloader_calldata != other.bootloader_calldata
558            || self.bootloader_heap != other.bootloader_heap
559            || self.bootloader_aux_heap != other.bootloader_aux_heap
560        {
561            return false;
562        }
563
564        for idx in 0..self.dynamic.len().max(other.dynamic.len()) {
565            match (self.dynamic.get(idx), other.dynamic.get(idx)) {
566                (Some(this_group), Some(other_group)) => {
567                    if this_group != other_group {
568                        return false;
569                    }
570                }
571                (Some(group), None) | (None, Some(group)) => {
572                    if !group.is_empty() {
573                        return false;
574                    }
575                }
576                (None, None) => {}
577            }
578        }
579        true
580    }
581}

583fn dynamic_slot_mut(
584    dynamic: &mut Vec<DynamicPageGroup>,
585    group: usize,
586    kind: DynamicPageKind,

588    if dynamic.len() <= group {

616impl PagePool {
617    fn allocate_page(&mut self) -> HeapPage {

619            .map(|mut page| {

621                page

628    }
629
630    fn recycle_page(&mut self, page: HeapPage) {
631        self.0.push(page);
632    }

644    fn test_heap_write_resizes(recycled_pages: &mut PagePool) {
645        let mut heap = Heap::default();
646        heap.write_u256(5, 1.into(), recycled_pages);
647        assert_eq!(heap.pages.len(), 1);
648        assert_eq!(heap.read_u256(5), 1.into());
649
650        // Check writing at a page boundary
651        heap.write_u256(
652            HEAP_PAGE_SIZE as u32 - 32,
653            repeat_byte(0xaa),
654            recycled_pages,
655        );
656        assert_eq!(heap.pages.len(), 1);
657        assert_eq!(
658            heap.read_u256(HEAP_PAGE_SIZE as u32 - 32),
659            repeat_byte(0xaa)
660        );
661
662        for offset in (1..=31).rev() {
663            heap.write_u256(
664                HEAP_PAGE_SIZE as u32 - offset,
665                repeat_byte(offset as u8),
666                recycled_pages,
667            );
668            assert_eq!(heap.pages.len(), 2);
669            assert_eq!(
670                heap.read_u256(HEAP_PAGE_SIZE as u32 - offset),
671                repeat_byte(offset as u8)
672            );
673        }
674
675        // check reading at a page boundary from a missing page
676        for offset in 0..32 {
677            assert_eq!(heap.read_u256((1 << 20) - offset), 0.into());
678        }
679
680        heap.write_u256(1 << 20, repeat_byte(0xff), recycled_pages);
681        assert_eq!(heap.pages.len(), 257);
682        assert_eq!(heap.pages.iter().flatten().count(), 3);
683        assert_eq!(heap.read_u256(1 << 20), repeat_byte(0xff));

744        let bytes: Vec<_> = (1..=32).collect();

746        for length in 1..=32 {

826    #[test]
827    fn rolling_back_heaps() {
828        let mut heaps = Heaps::new(b"test");
829        let written_value = U256::from(123_456_789) << 224; // writes bytes 0..4
830        heaps.write_u256(HeapId::FIRST, 0, written_value);
831        assert_eq!(heaps[HeapId::FIRST].read_u256(0), written_value);
832        heaps.write_u256(HeapId::FIRST_AUX, 0, 42.into());
833        assert_eq!(heaps[HeapId::FIRST_AUX].read_u256(0), 42.into());
834
835        let snapshot = heaps.snapshot();
836        assert_eq!(snapshot, (1, 1));
837
838        heaps.write_u256(HeapId::FIRST, 7, U256::MAX);
839        assert_eq!(
840            heaps[HeapId::FIRST].read_u256(0),
841            written_value + (U256::MAX >> 56)
842        );
843        heaps.write_u256(HeapId::FIRST_AUX, 16, U256::MAX);
844        assert_eq!(heaps[HeapId::FIRST_AUX].read_u256(16), U256::MAX);
845
846        heaps.rollback(snapshot);
847        assert_eq!(heaps[HeapId::FIRST].read_u256(0), written_value);
848        assert_eq!(heaps[HeapId::FIRST_AUX].read_u256(0), 42.into());
849        assert_eq!(heaps.bootloader_heap_rollback_info.len(), 1);
850        assert_eq!(heaps.bootloader_aux_rollback_info.len(), 1);
851    }
