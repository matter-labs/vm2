13/// Ensures that a decommit hash has a materialized reusable page and returns it.
14///
15/// The resulting page is pinned globally in [`WorldDiff`]. If that page is not owned by the
16/// current frame, it is also recorded as kept alive in the bootloader frame (or current frame if
17/// no bootloader frame exists), matching decommit opcode teardown semantics.
18pub(crate) fn materialize_decommit_page<T: Tracer, W: World<T>>(
19    vm: &mut VirtualMachine<T, W>,
20    code_hash: U256,
21    code: &[u8],
22    candidate_page: HeapId,
23) -> HeapId {
24    if let Some(existing) = vm.world_diff.decommit_page(code_hash) {
25        return existing;
26    }
27
28    vm.state.heaps.write_bytes(candidate_page, 0, code);
29    let heap = candidate_page;
30    vm.world_diff.set_decommit_page(code_hash, heap);
31
32    if heap != vm.state.current_frame.heap && heap != vm.state.current_frame.aux_heap {
33        let heaps_to_keep_alive =
34            if let Some(bootloader_frame) = vm.state.previous_frames.first_mut() {
35                &mut bootloader_frame.heaps_i_am_keeping_alive
36            } else {
37                &mut vm.state.current_frame.heaps_i_am_keeping_alive
38            };
39        heaps_to_keep_alive.push(heap);
40    }
41
42    heap
43}

47    pub(crate) fn decommit<T: Tracer>(
48        &mut self,
49        world: &mut impl World<T>,
50        tracer: &mut T,
51        address: U256,
52        default_aa_code_hash: [u8; 32],
53        evm_interpreter_code_hash: [u8; 32],
54        is_constructor_call: bool,
55        tx_number_in_block: u16,
56    ) -> Option<(UnpaidDecommit, bool, bool)> {
57        let deployer_system_contract_address =
58            Address::from_low_u64_be(DEPLOYER_SYSTEM_CONTRACT_ADDRESS_LOW.into());
59
60        let mut is_evm = false;
61        // zk_evm grants the larger EVM frame stipend based on the version byte alone.
62        // Keep this separate from `is_evm`: calls masked to default AA should not execute
63        // via the EVM interpreter, but their frame bounds still expose the original blob
64        // format classification.
65        let mut is_evm_blob_format = false;
66
67        let mut code_info = {
68            let code_info = self.read_storage_without_refund(
69                world,
70                tracer,
71                deployer_system_contract_address,
72                address,
73                tx_number_in_block,
74            );
75            let mut code_info_bytes = [0; 32];
76            code_info.to_big_endian(&mut code_info_bytes);
77
78            // Note that EOAs are considered constructed because their code info is all zeroes.
79            let is_constructed = match code_info_bytes[1] {
80                0 => true,
81                1 => false,
82                _ => {
83                    return None;
84                }
85            };
86
87            let try_default_aa = if is_kernel(u256_into_address(address)) {
88                None
89            } else {
90                Some(default_aa_code_hash)
91            };
92
93            // The address aliasing contract implements Ethereum-like behavior of calls to EOAs
94            // returning successfully (and address aliasing when called from the bootloader).
95            // It makes sense that unconstructed code is treated as an EOA but for some reason
96            // a constructor call to constructed code is also treated as EOA.
97            match code_info_bytes[0] {
98                1 => {
99                    if is_constructed == is_constructor_call {
100                        try_default_aa?
101                    } else {
102                        code_info_bytes
103                    }
104                }
105                2 => {
106                    is_evm_blob_format = true;
107                    if is_constructed == is_constructor_call {
108                        try_default_aa?
109                    } else {
110                        is_evm = true;
111                        evm_interpreter_code_hash
112                    }
113                }
114                _ if code_info == U256::zero() => try_default_aa?,
115                _ => return None,
116            }
117        };
118
119        code_info[1] = 0;
120        let code_key: U256 = U256::from_big_endian(&code_info);
121
122        let decommit_state = self.decommitted_hashes.as_ref().get(&code_key).copied();
123        let was_decommitted = matches!(decommit_state, Some(DecommitState::Succeeded(_)));
124        let cost = if was_decommitted {
125            0
126        } else {
127            let code_length_in_words = u16::from_be_bytes([code_info[2], code_info[3]]);
128            u32::from(code_length_in_words) * zkevm_opcode_defs::ERGS_PER_CODE_WORD_DECOMMITTMENT
129        };
130
131        let should_materialize = !was_decommitted;
132
133        Some((
134            UnpaidDecommit {
135                cost,
136                code_key,
137                should_materialize,
138            },
139            is_evm,
140            is_evm_blob_format,
141        ))
142    }
143
144    /// Returns the decommitted contract code and a flag set to `true` if this is a fresh decommit (i.e.,
145    /// the code wasn't decommitted previously in the same VM run).
146    #[doc(hidden)] // should be used for testing purposes only; can break VM operation otherwise
147    pub fn decommit_opcode<T: Tracer>(
148        &mut self,
149        world: &mut impl World<T>,
150        tracer: &mut T,
151        code_hash: U256,
152    ) -> (Vec<u8>, bool) {
153        let is_new = match self.decommitted_hashes.as_ref().get(&code_hash).copied() {
154            None | Some(DecommitState::Unsuccessful) => true,
155            Some(DecommitState::Succeeded(_)) => false,
156        };
157        let code = world.decommit_code(code_hash);
158        if is_new {
159            let code_len = u32::try_from(code.len()).expect("bytecode length overflow");
160            // Decommitter can process two words per cycle, hence division by 2 * 32 = 64.
161            tracer.on_extra_prover_cycles(CycleStats::Decommit(code_len.div_ceil(64)));
162        }
163        (code, is_new)
164    }

166    pub(crate) fn pay_for_decommit<T: Tracer, W: World<T>>(
167        &mut self,
168        world: &mut W,
169        tracer: &mut T,
170        decommit: UnpaidDecommit,
171        gas: &mut u32,
172    ) -> Option<Program<T, W>> {
173        if decommit.cost > *gas {
174            // We intentionally keep this hash visible even though decommit did not execute.
175            // Legacy VM includes far-call OOG attempts in "used contracts", and shadow mode
176            // compares that output.
177            if !matches!(
178                self.decommitted_hashes.as_ref().get(&decommit.code_key),
179                Some(DecommitState::Succeeded(_))
180            ) {
181                self.decommitted_hashes
182                    .insert(decommit.code_key, DecommitState::Unsuccessful);
183            }
184            // Unlike all other gas costs, this one is not paid if low on gas.
185            return None;
186        }
187
188        let is_new = match self
189            .decommitted_hashes
190            .as_ref()
191            .get(&decommit.code_key)
192            .copied()
193        {
194            None | Some(DecommitState::Unsuccessful) => true,
195            Some(DecommitState::Succeeded(_)) => false,
196        };
197        *gas -= decommit.cost;
198
199        let decommit = world.decommit(decommit.code_key);
200        if is_new {
201            let code_len_in_words =
202                u32::try_from(decommit.code_page().len()).expect("bytecode length overflow");
203            // Decommitter can process two words per cycle.
204            tracer.on_extra_prover_cycles(CycleStats::Decommit(code_len_in_words.div_ceil(2)));
205        }
206
207        Some(decommit)
208    }

229pub(crate) fn u256_into_address(source: U256) -> H160 {
230    let mut result = H160::zero();
231    let mut bytes = [0; 32];
232    source.to_big_endian(&mut bytes);
233    result.assign_from_slice(&bytes[12..]);
234    result
235}

237pub(crate) fn is_kernel(address: H160) -> bool {
