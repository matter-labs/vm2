27#[allow(clippy::too_many_lines)]
28pub(crate) fn decode<T: Tracer, W: World<T>>(raw: u64, is_bootloader: bool) -> Instruction<T, W> {
29    let (parsed, _) = EncodingModeProduction::parse_preliminary_variant_and_absolute_number(raw);
30
31    let predicate = match parsed.condition {
32        zkevm_opcode_defs::Condition::Always => Predicate::Always,
33        zkevm_opcode_defs::Condition::Gt => Predicate::IfGT,
34        zkevm_opcode_defs::Condition::Lt => Predicate::IfLT,
35        zkevm_opcode_defs::Condition::Eq => Predicate::IfEQ,
36        zkevm_opcode_defs::Condition::Ge => Predicate::IfGE,
37        zkevm_opcode_defs::Condition::Le => Predicate::IfLE,
38        zkevm_opcode_defs::Condition::Ne => Predicate::IfNotEQ,
39        zkevm_opcode_defs::Condition::GtOrLt => Predicate::IfGTOrLT,
40    };
41    let arguments = Arguments::new(
42        predicate,
43        parsed.variant.ergs_price(),
44        ModeRequirements::new(
45            parsed.variant.requires_kernel_mode(),
46            !parsed.variant.can_be_used_in_static_context(),
47        ),
48    );
49
50    let stack_in = RegisterAndImmediate {
51        immediate: parsed.imm_0,
52        register: Register::new(parsed.src0_reg_idx),
53    };
54    let src1: AnySource = match parsed.variant.src0_operand_type {
55        RegOnly | RegOrImm(RegOrImmFlags::UseRegOnly) | Full(ImmMemHandlerFlags::UseRegOnly) => {
56            Register1(Register::new(parsed.src0_reg_idx)).into()
57        }
58        RegOrImm(RegOrImmFlags::UseImm16Only) | Full(ImmMemHandlerFlags::UseImm16Only) => {
59            Immediate1(parsed.imm_0).into()
60        }
61        Full(ImmMemHandlerFlags::UseAbsoluteOnStack) => AbsoluteStack(stack_in).into(),
62        Full(ImmMemHandlerFlags::UseStackWithPushPop) => AdvanceStackPointer(stack_in).into(),
63        Full(ImmMemHandlerFlags::UseStackWithOffset) => RelativeStack(stack_in).into(),
64        Full(ImmMemHandlerFlags::UseCodePage) => CodePage(stack_in).into(),
65    };
66
67    let stack_out = RegisterAndImmediate {
68        immediate: parsed.imm_1,
69        register: Register::new(parsed.dst0_reg_idx),
70    };
71    let out: AnyDestination = match parsed.variant.dst0_operand_type {
72        RegOnly | RegOrImm(RegOrImmFlags::UseRegOnly) | Full(ImmMemHandlerFlags::UseRegOnly) => {
73            Register1(Register::new(parsed.dst0_reg_idx)).into()
74        }
75        RegOrImm(RegOrImmFlags::UseImm16Only) | Full(ImmMemHandlerFlags::UseImm16Only) => {
76            panic!("Parser wants to output to immediate")
77        }
78        Full(ImmMemHandlerFlags::UseAbsoluteOnStack) => AbsoluteStack(stack_out).into(),
79        Full(ImmMemHandlerFlags::UseStackWithPushPop) => AdvanceStackPointer(stack_out).into(),
80        Full(ImmMemHandlerFlags::UseStackWithOffset) => RelativeStack(stack_out).into(),
81        Full(ImmMemHandlerFlags::UseCodePage) => panic!("Parser wants to write to code page"),
82    };
83
84    let src2 = Register2(Register::new(parsed.src1_reg_idx));
85    let out2 = Register2(Register::new(parsed.dst1_reg_idx));
86
87    macro_rules! binop {
88        ($op: ident, $snd: tt) => {
89            Instruction::from_binop::<$op>(
90                src1,
91                src2,
92                out,
93                &$snd,
94                arguments,
95                parsed.variant.flags[SWAP_OPERANDS_FLAG_IDX_FOR_ARITH_OPCODES],
96                parsed.variant.flags[SET_FLAGS_FLAG_IDX],
97            )
98        };
99    }
100
101    macro_rules! ptr {
102        ($op: ident) => {
103            Instruction::from_ptr::<$op>(
104                src1,
105                src2,
106                out,
107                arguments,
108                parsed.variant.flags[SWAP_OPERANDS_FLAG_IDX_FOR_PTR_OPCODE],
109            )
110        };
111    }
112
113    match parsed.variant.opcode {
114        Opcode::Add(_) => binop!(Add, ()),
115        Opcode::Sub(_) => binop!(Sub, ()),
116        Opcode::Mul(_) => binop!(Mul, out2),
117        Opcode::Div(_) => binop!(Div, out2),
118        Opcode::Binop(x) => match x {
119            zkevm_opcode_defs::BinopOpcode::Xor => binop!(Xor, ()),
120            zkevm_opcode_defs::BinopOpcode::And => binop!(And, ()),
121            zkevm_opcode_defs::BinopOpcode::Or => binop!(Or, ()),
122        },
123        Opcode::Shift(x) => match x {
124            zkevm_opcode_defs::ShiftOpcode::Shl => binop!(ShiftLeft, ()),
125            zkevm_opcode_defs::ShiftOpcode::Shr => binop!(ShiftRight, ()),
126            zkevm_opcode_defs::ShiftOpcode::Rol => binop!(RotateLeft, ()),
127            zkevm_opcode_defs::ShiftOpcode::Ror => binop!(RotateRight, ()),
128        },
129        Opcode::Jump(_) => Instruction::from_jump(src1, out.try_into().unwrap(), arguments),
130        Opcode::Context(x) => match x {
131            zkevm_opcode_defs::ContextOpcode::This => {
132                Instruction::from_this(out.try_into().unwrap(), arguments)
133            }
134            zkevm_opcode_defs::ContextOpcode::Caller => {
135                Instruction::from_caller(out.try_into().unwrap(), arguments)
136            }
137            zkevm_opcode_defs::ContextOpcode::CodeAddress => {
138                Instruction::from_code_address(out.try_into().unwrap(), arguments)
139            }
140            zkevm_opcode_defs::ContextOpcode::ErgsLeft => {
141                Instruction::from_ergs_left(out.try_into().unwrap(), arguments)
142            }
143            zkevm_opcode_defs::ContextOpcode::GetContextU128 => {
144                Instruction::from_context_u128(out.try_into().unwrap(), arguments)
145            }
146            zkevm_opcode_defs::ContextOpcode::SetContextU128 => {
147                Instruction::from_set_context_u128(src1.try_into().unwrap(), arguments)
148            }
149            zkevm_opcode_defs::ContextOpcode::Sp => {
150                Instruction::from_context_sp(out.try_into().unwrap(), arguments)
151            }
152            zkevm_opcode_defs::ContextOpcode::Meta => {
153                Instruction::from_context_meta(out.try_into().unwrap(), arguments)
154            }
155            zkevm_opcode_defs::ContextOpcode::IncrementTxNumber => {
156                Instruction::from_increment_tx_number(arguments)
157            }
158            zkevm_opcode_defs::ContextOpcode::AuxMutating0 => {
159                Instruction::from_aux_mutating(arguments)
160            }
161        },
162        Opcode::Ptr(x) => match x {
163            zkevm_opcode_defs::PtrOpcode::Add => ptr!(PointerAdd),
164            zkevm_opcode_defs::PtrOpcode::Sub => ptr!(PointerSub),
165            zkevm_opcode_defs::PtrOpcode::Pack => ptr!(PointerPack),
166            zkevm_opcode_defs::PtrOpcode::Shrink => ptr!(PointerShrink),
167        },
168        Opcode::NearCall(_) => Instruction::from_near_call(
169            Register1(Register::new(parsed.src0_reg_idx)),
170            Immediate1(parsed.imm_0),
171            Immediate2(parsed.imm_1),
172            arguments,
173        ),
174        Opcode::FarCall(kind) => {
175            let constructor = match kind {
176                zkevm_opcode_defs::FarCallOpcode::Normal => {
177                    Instruction::from_far_call::<opcodes::Normal>
178                }
179                zkevm_opcode_defs::FarCallOpcode::Delegate => {
180                    Instruction::from_far_call::<opcodes::Delegate>
181                }
182                zkevm_opcode_defs::FarCallOpcode::Mimic => {
183                    Instruction::from_far_call::<opcodes::Mimic>
184                }
185            };
186            constructor(
187                src1.try_into().unwrap(),
188                src2,
189                Immediate1(parsed.imm_0),
190                parsed.variant.flags[FAR_CALL_STATIC_FLAG_IDX],
191                parsed.variant.flags[FAR_CALL_SHARD_FLAG_IDX],
192                arguments,
193            )
194        }
195        Opcode::Ret(kind) => {
196            let to_label = parsed.variant.flags[RET_TO_LABEL_BIT_IDX];
197            let label = if to_label {
198                Some(Immediate1(parsed.imm_0))
199            } else {
200                None
201            };
202            match kind {
203                zkevm_opcode_defs::RetOpcode::Ok => {
204                    Instruction::from_ret(src1.try_into().unwrap(), label, arguments)
205                }
206                zkevm_opcode_defs::RetOpcode::Revert => {
207                    Instruction::from_revert(src1.try_into().unwrap(), label, arguments)
208                }
209                zkevm_opcode_defs::RetOpcode::Panic => Instruction::from_panic(label, arguments),
210            }
211        }
212        Opcode::Log(x) => match x {
213            zkevm_opcode_defs::LogOpcode::StorageRead => Instruction::from_storage_read(
214                src1.try_into().unwrap(),
215                out.try_into().unwrap(),
216                arguments,
217            ),
218            zkevm_opcode_defs::LogOpcode::TransientStorageRead => {
219                Instruction::from_transient_storage_read(
220                    src1.try_into().unwrap(),
221                    out.try_into().unwrap(),
222                    arguments,
223                )
224            }
225
226            zkevm_opcode_defs::LogOpcode::StorageWrite => {
227                Instruction::from_storage_write(src1.try_into().unwrap(), src2, arguments)
228            }
229
230            zkevm_opcode_defs::LogOpcode::TransientStorageWrite => {
231                Instruction::from_transient_storage_write(src1.try_into().unwrap(), src2, arguments)
232            }
233
234            zkevm_opcode_defs::LogOpcode::ToL1Message => Instruction::from_l2_to_l1_message(
235                src1.try_into().unwrap(),
236                src2,
237                parsed.variant.flags[FIRST_MESSAGE_FLAG_IDX],
238                arguments,
239            ),
240            zkevm_opcode_defs::LogOpcode::Event => Instruction::from_event(
241                src1.try_into().unwrap(),
242                src2,
243                parsed.variant.flags[FIRST_MESSAGE_FLAG_IDX],
244                arguments,
245            ),
246            zkevm_opcode_defs::LogOpcode::PrecompileCall => Instruction::from_precompile_call(
247                src1.try_into().unwrap(),
248                src2,
249                out.try_into().unwrap(),
250                arguments,
251            ),
252            zkevm_opcode_defs::LogOpcode::Decommit => Instruction::from_decommit(
253                src1.try_into().unwrap(),
254                src2,
255                out.try_into().unwrap(),
256                arguments,
257            ),
258        },
259        Opcode::UMA(x) => {
260            let increment = parsed.variant.flags[UMA_INCREMENT_FLAG_IDX];
261            match x {
262                zkevm_opcode_defs::UMAOpcode::HeapRead => Instruction::from_heap_read(
263                    src1.try_into().unwrap(),
264                    out.try_into().unwrap(),
265                    increment.then_some(out2),
266                    arguments,
267                ),
268                zkevm_opcode_defs::UMAOpcode::HeapWrite => Instruction::from_heap_write(
269                    src1.try_into().unwrap(),
270                    src2,
271                    increment.then_some(out.try_into().unwrap()),
272                    arguments,
273                    is_bootloader,
274                ),
275                zkevm_opcode_defs::UMAOpcode::AuxHeapRead => Instruction::from_aux_heap_read(
276                    src1.try_into().unwrap(),
277                    out.try_into().unwrap(),
278                    increment.then_some(out2),
279                    arguments,
280                ),
281                zkevm_opcode_defs::UMAOpcode::AuxHeapWrite => Instruction::from_aux_heap_store(
282                    src1.try_into().unwrap(),
283                    src2,
284                    increment.then_some(out.try_into().unwrap()),
285                    arguments,
286                ),
287                zkevm_opcode_defs::UMAOpcode::FatPointerRead => Instruction::from_pointer_read(
288                    src1.try_into().unwrap(),
289                    out.try_into().unwrap(),
290                    increment.then_some(out2),
291                    arguments,
292                ),
293                zkevm_opcode_defs::UMAOpcode::StaticMemoryRead => {
294                    Instruction::from_static_memory_read(
295                        src1.try_into().unwrap(),
296                        out.try_into().unwrap(),
297                        increment.then_some(out2),
298                        arguments,
299                    )
300                }
301                zkevm_opcode_defs::UMAOpcode::StaticMemoryWrite => {
302                    Instruction::from_static_memory_write(
303                        src1.try_into().unwrap(),
304                        src2,
305                        increment.then_some(out.try_into().unwrap()),
306                        arguments,
307                    )
308                }
309            }
310        }
311        Opcode::Invalid(_) => Instruction::from_invalid(),
312        Opcode::Nop(_) => {
313            let no_sp_movement = AdvanceStackPointer(RegisterAndImmediate {
314                immediate: 0,
315                register: Register::new(0),
316            });
317            Instruction::from_nop(
318                if let AnySource::AdvanceStackPointer(pop) = src1 {
319                    pop
320                } else {
321                    no_sp_movement
322                },
323                if let AnyDestination::AdvanceStackPointer(push) = out {
324                    push
325                } else {
326                    no_sp_movement
327                },
328                arguments,
329            )
330        }
331    }
332}
