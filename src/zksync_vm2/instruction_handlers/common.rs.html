36#[inline(always)]
37pub(crate) fn full_boilerplate<Opcode: OpcodeType, T: Tracer, W: World<T>>(
38    vm: &mut VirtualMachine<T, W>,
39    world: &mut W,
40    tracer: &mut T,
41    business_logic: impl FnOnce(
42        &mut VirtualMachine<T, W>,
43        &Arguments,
44        &mut W,
45        &mut T,
46    ) -> ExecutionStatus,
47) -> ExecutionStatus {
48    let args = unsafe { &(*vm.state.current_frame.pc).arguments };
49
50    // We are not checking whether the callstack is full here. However,
51    // the callstack depth limit on EraVM is so high that it will not be hit in practice,
52    // so it is fine to not have this check, even though `zk_evm` has it.
53    // Introducing this check would require additional bookkeeping for far/near calls, which
54    // doesn't currently make sense.
55
56    // Make sure that call stack depth is very high, just in case.
57    #[allow(clippy::items_after_statements)] // Invariant for the code below
58    const _: () = assert!(VM_MAX_STACK_DEPTH >= 214_748_444);
59
60    if vm.state.use_gas(args.get_static_gas_cost()).is_err()
61        || !args.mode_requirements().met(
62            vm.state.current_frame.is_kernel,
63            vm.state.current_frame.is_static,
64        )
65    {
66        return free_panic(vm, world, tracer);
67    }
68
69    if args.predicate().satisfied(&vm.state.flags) {
70        // Capture the second-output register before running the instruction: business logic may pop
71        // the current frame (e.g. `ret`/`far_call`) and free the code page `args` points into, so it
72        // must not be dereferenced afterwards. `Register` is a plain index and stays valid.
73        let dst1 = args.dst1_register();
74        vm.state.dst1_was_updated = false;
75
76        tracer.before_instruction::<Opcode, _>(&mut VmAndWorld { vm, world });
77        vm.state.current_frame.pc = unsafe { vm.state.current_frame.pc.add(1) };
78        let result = business_logic(vm, args, world, tracer);
79
80        // Unless the instruction wrote its second output, `zk_evm` leaves `dst1` cleared to zero.
81        // Do the same here, after the primary output so that a `dst0`/`dst1` alias resolves to zero.
82        if !vm.state.dst1_was_updated {
83            dst1.set_zero(&mut vm.state);
84        }
85
86        result.merge_tracer(tracer.after_instruction::<Opcode, _>(&mut VmAndWorld { vm, world }))
87    } else {
88        tracer.before_instruction::<opcodes::Nop, _>(&mut VmAndWorld { vm, world });
89        vm.state.current_frame.pc = unsafe { vm.state.current_frame.pc.add(1) };
90        tracer
91            .after_instruction::<opcodes::Nop, _>(&mut VmAndWorld { vm, world })
92            .into()
93    }
94}
