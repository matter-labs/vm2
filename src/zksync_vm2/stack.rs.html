10#[derive(PartialEq)]
11pub(crate) struct Stack {
12    /// set of slots that may be interpreted as [`FatPointer`].
13    pointer_flags: Bitset,
14    dirty_areas: u64,
15    slots: [U256; 1 << 16],
16}
17
18const NUMBER_OF_DIRTY_AREAS: usize = 64;
19const DIRTY_AREA_SIZE: usize = (1 << 16) / NUMBER_OF_DIRTY_AREAS;
20
21impl Stack {
22    #[allow(clippy::cast_ptr_alignment)] // aligned per `Stack` layout
23    pub(crate) fn new() -> Box<Self> {
24        unsafe { Box::from_raw(alloc_zeroed(Layout::new::<Self>()).cast()) }
25    }
26
27    #[inline(always)]
28    pub(crate) fn get(&self, slot: u16) -> U256 {
29        self.slots[slot as usize]
30    }
31
32    #[inline(always)]
33    pub(crate) fn set(&mut self, slot: u16, value: U256) {
34        let written_area = slot as usize / DIRTY_AREA_SIZE;
35        self.dirty_areas |= 1 << written_area;
36
37        self.slots[slot as usize] = value;
38    }
39
40    fn zero(&mut self) {
41        for i in 0..NUMBER_OF_DIRTY_AREAS {
42            if self.dirty_areas & (1 << i) != 0 {
43                for slot in &mut self.slots[i * DIRTY_AREA_SIZE..(i + 1) * DIRTY_AREA_SIZE] {
44                    *slot = U256::zero();
45                }
46            }
47        }
48
49        self.dirty_areas = 0;
50        self.pointer_flags = Bitset::default();
51    }
52
53    #[inline(always)]
54    pub(crate) fn get_pointer_flag(&self, slot: u16) -> bool {
55        self.pointer_flags.get(slot)
56    }
57
58    #[inline(always)]
59    pub(crate) fn set_pointer_flag(&mut self, slot: u16) {
60        self.pointer_flags.set(slot);
61    }
62
63    #[inline(always)]
64    pub(crate) fn clear_pointer_flag(&mut self, slot: u16) {
65        self.pointer_flags.clear(slot);
66    }
67
68    pub(crate) fn snapshot(&self) -> StackSnapshot {
69        let dirty_prefix_end = NUMBER_OF_DIRTY_AREAS - self.dirty_areas.leading_zeros() as usize;
70
71        StackSnapshot {
72            pointer_flags: self.pointer_flags.clone(),
73            dirty_areas: self.dirty_areas,
74            slots: self.slots[..DIRTY_AREA_SIZE * dirty_prefix_end].into(),
75        }
76    }
77
78    pub(crate) fn rollback(&mut self, snapshot: StackSnapshot) {
79        let StackSnapshot {
80            pointer_flags,
81            dirty_areas,
82            slots,
83        } = snapshot;
84
85        self.zero();
86
87        self.pointer_flags = pointer_flags;
88        self.dirty_areas = dirty_areas;
89        self.slots[..slots.len()].copy_from_slice(&slots);
90    }
91}

99#[allow(clippy::borrow_as_ptr)]
100impl Clone for Box<Stack> {
101    fn clone(&self) -> Self {
102        unsafe {
103            let allocation = alloc(Layout::for_value(&**self)).cast();
104            std::ptr::copy_nonoverlapping(&**self, allocation, 1);
105            Box::from_raw(allocation)
106        }
107    }
108}

117        self.stacks.pop().map_or_else(Stack::new, |mut s| {
118            s.zero();
119            s
120        })
121    }

130/// Helper wrapper for debugging [`Stack`] / [`StackSnapshot`] contents.
131struct StackStart<I>(I);
132
133impl<I: Iterator<Item = (bool, U256)> + Clone> fmt::Debug for StackStart<I> {
134    fn fmt(&self, formatter: &mut fmt::Formatter<'_>) -> fmt::Result {
135        let mut list = formatter.debug_list();
136        for (is_pointer, slot) in self.0.clone() {
137            if is_pointer {
138                list.entry(&FatPointer::from(slot));
139            } else {
140                list.entry(&slot);
141            }
142        }
143        list.finish()
144    }
145}
146
147impl fmt::Debug for Stack {
148    fn fmt(&self, formatter: &mut fmt::Formatter<'_>) -> fmt::Result {
149        const DEBUGGED_SLOTS: usize = 256;
150
151        let slots = (0_u16..)
152            .zip(&self.slots)
153            .map(|(idx, slot)| (self.pointer_flags.get(idx), *slot))
154            .take(DEBUGGED_SLOTS);
155        formatter
156            .debug_struct("Stack")
157            .field("start", &StackStart(slots))
158            .field(
159                "pointer_flags.hash",
160                &hash_for_debugging(&self.pointer_flags),
161            )
162            .field("slots.hash", &hash_for_debugging(&self.slots))
163            .finish_non_exhaustive()
164    }
165}
166
167impl fmt::Debug for StackSnapshot {
168    fn fmt(&self, formatter: &mut fmt::Formatter<'_>) -> fmt::Result {
169        const DEBUGGED_SLOTS: usize = 256;
170
171        let slots = (0_u16..)
172            .zip(&self.slots[..])
173            .map(|(idx, slot)| (self.pointer_flags.get(idx), *slot))
174            .take(DEBUGGED_SLOTS);
175        formatter
176            .debug_struct("StackSnapshot")
177            .field("dirty_areas", &self.dirty_areas)
178            .field("start", &StackStart(slots))
179            .field(
180                "pointer_flags.hash",
181                &hash_for_debugging(&self.pointer_flags),
182            )
183            .field("slots.hash", &hash_for_debugging(&self.slots))
184            .finish_non_exhaustive()
185    }
186}
